//代码片段
i#include <bits/stdc++.h>
using namespace std;

void quicksort(int a[],int left,int right) {
	int i=left,j=right;
	if(left>right)
		return ;
	i=left;
	j=right;
	while(i!=j) {
		while(a[j]>=a[left]&&i<j)
			j--;
		while(a[i]<=a[left]&&i<j)
			i++;
		//交换两个数在数组中的位置
		if(i<j)  //没有相遇时
			swap(a[i], a[j]);
	}
	swap(a[left], a[i]);
	quicksort(a,left,i-1);//继续处理左边的，这里是一个递归的过程
	quicksort(a,i+1,right);//继续处理右边的，这里是一个递归的过程
}
int main() {
	int a[101000];
	int n;
	cin>>n;
	int i;
	for(i=1; i<=n; i++)
		cin>>a[i];
	quicksort(a,1,n);
	for(i=1; i<=n; i++) {
		cout<<a[i]<<" ";
	}
    cout << endl;
    return 0;
}
