List Reversr (List L){
    List p,temp;
    P->Next =NULL;//给p的指针域赋空作为链表尾部
    while(L){
        L->Next=temp;//先处理第一个结点，就先把后面的结点都先存下来
        L->Next=p;//将第一个结点指向p
        p=L;//再将p指针变成头部
        L=temp;//把预先保留的链表剩余部分赋回去
    }
    return p;//进行完循环，p仍是指针头部，返回p即可
}
